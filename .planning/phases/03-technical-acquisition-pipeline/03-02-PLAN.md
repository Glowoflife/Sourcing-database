import { Readability } from "@mozilla/readability";
import TurndownService from "turndown";
import { gfm } from "turndown-plugin-gfm";

const turndown = new TurndownService({ headingStyle: "atx" });
turndown.use(gfm); // Enables Markdown tables (D-04)

export function htmlToMarkdown(html: string, url: string): string {
  const dom = new JSDOM(html, { url });
  const reader = new Readability(dom.window.document);
  const article = reader.parse();
  if (!article) return turndown.turndown(html); // fallback for SPA/thin pages
  return turndown.turndown(article.content);    // MUST use article.content, not article.textContent
}

  **File 2: src/acquisition/html-to-markdown.ts**

  The TurndownService instance is created once at module level (singleton) — TurndownService is stateless between calls and creating it per-call is wasteful.

  CRITICAL RULE: Always pass `article.content` to `turndown.turndown()`, NEVER `article.textContent`.
  - `article.content` = cleaned HTML (tables, headings, lists preserved — Turndown can convert these)
  - `article.textContent` = plain text with ALL HTML tags stripped (tables become flat whitespace — violates D-04)

  ```typescript
  import { JSDOM } from "jsdom";
  import { Readability } from "@mozilla/readability";
  import TurndownService from "turndown";
  import { gfm } from "turndown-plugin-gfm";

  // Create once at module level — TurndownService is stateless between conversions
  const turndown = new TurndownService({ headingStyle: "atx" });
  turndown.use(gfm); // D-04: enables <table> → Markdown table output

  /**
   * Convert a raw HTML string to clean Markdown.
   *
   * @param html  Raw HTML content from page.content() (Playwright)
   * @param url   The page's URL — required by jsdom for correct relative-URL resolution
   * @returns     Markdown string; falls back to raw-HTML Turndown if Readability cannot parse
   */
  export function htmlToMarkdown(html: string, url: string): string {
    // jsdom provides the DOM environment that Readability requires in Node.js
    const dom = new JSDOM(html, { url });
    const reader = new Readability(dom.window.document);
    const article = reader.parse();

    if (!article) {
      // Readability could not extract readable content (SPA, thin page, pure JS shell).
      // Fall back to converting raw HTML — still better than storing raw HTML for Phase 4.
      return turndown.turndown(html);
    }

    // MUST use article.content (cleaned HTML with structure intact).
    // DO NOT use article.textContent — that is plain text with all tags stripped,
    // which would flatten all tables to whitespace, violating D-04.
    return turndown.turndown(article.content);
  }
  ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| REDIS_URL (env) → IORedis constructor | Connection string from environment controls which Redis instance is connected |
| Manufacturer HTML → Readability → Turndown → Markdown string | Untrusted website HTML enters the conversion pipeline |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04 | Tampering | Readability parse() applied to adversarial HTML (e.g., crafted manufacturer page with malicious content) | accept | Readability and jsdom run in a Node.js child process context; the output is a Markdown string stored in DB. No user-facing rendering occurs in Phase 3 — Phase 5 will sanitize before display. |
| T-03-05 | Information Disclosure | REDIS_URL leakage via error messages thrown by redis.ts env-guard | accept | Error message includes the variable name but not the value; thrown at startup, visible only to developers running the CLI |
| T-03-06 | Denial of Service | Large HTML pages (e.g., 5MB) passed to jsdom/Readability consuming excessive memory | accept | Manufacturer pages are typically small (<500KB); each BullMQ job processes one page at a time; Crawlee's requestHandlerTimeoutSecs: 60 limits per-page time. Monitor in Phase 3 smoke test — if memory is a concern, add a size cap in Phase 4. |
