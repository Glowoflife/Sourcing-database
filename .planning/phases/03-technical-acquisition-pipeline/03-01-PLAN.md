export const leads = pgTable("leads", {
  id: serial("id").primaryKey(),
  name: text("name").notNull(),
  url: text("url").unique(),
  status: leadStatusEnum("status").notNull().default("New"),
  createdAt: timestamp("created_at").notNull().defaultNow(),
  updatedAt: timestamp("updated_at").notNull().defaultNow().$onUpdateFn(() => new Date()),

  id: serial("id").primaryKey(),
  startedAt: timestamp("started_at").notNull().defaultNow(),
  finishedAt: timestamp("finished_at"),
  leadsFound: integer("leads_found").notNull().default(0),
  leadsWritten: integer("leads_written").notNull().default(0),
  leadsSkipped: integer("leads_skipped").notNull().default(0),
  leadsErrored: integer("leads_errored").notNull().default(0),
});

export type ScraperRun = typeof scraperRuns.$inferSelect;
export type NewScraperRun = typeof scraperRuns.$inferInsert;

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| .env.local → process.env | Secrets (DATABASE_URL, REDIS_URL) loaded from local file into Node.js environment |
| Node.js → PostgreSQL | drizzle-kit push creates schema objects; DATABASE_URL controls which database is modified |
| Node.js → Redis | REDIS_URL controls which Redis instance BullMQ connects to |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Information Disclosure | .env.local contains DATABASE_URL and REDIS_URL in plaintext | accept | Developer machine only; .env.local is in .gitignore (standard Next.js convention); no PII in connection strings |
| T-03-02 | Tampering | drizzle-kit push modifies the production database if DATABASE_URL points to production | mitigate | Push command reads DATABASE_URL from .env.local; developer is responsible for verifying the URL points to the dev/local DB before running push. Acceptance criteria includes verifying the DB after push. |
| T-03-03 | Denial of Service | Redis server accepts connections from any local process | accept | Redis runs on localhost only (no -p 0.0.0.0 binding); local dev machine; no auth required for local dev use |
</threat_model>
