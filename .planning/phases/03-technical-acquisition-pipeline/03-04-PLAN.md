  ```typescript
  import { Worker } from "bullmq";
  import type { Job } from "bullmq";
  import { redis } from "@/lib/redis";
  import { runAcquisitionJob } from "@/acquisition/index";
  import { logger } from "@/lib/logger";

  export const acquisitionWorker = new Worker(
    "acquisition", // D-06: must match Queue name in src/workers/queues.ts
    async (job: Job) => {
      const { leadId, url } = job.data as { leadId: number; url: string };
      await runAcquisitionJob({ leadId, url });
    },
    {
      connection: redis,
      concurrency: 3, // D-06: 3 concurrent Playwright workers
    },
  );

  acquisitionWorker.on("completed", (job: Job) => {
    logger.info({
      stage: "worker",
      status: "ok",
      leadId: job.data.leadId,
      message: `Job ${job.id} completed`,
    });
  });

  acquisitionWorker.on("failed", (job: Job | undefined, err: Error) => {
    logger.error({
      stage: "worker",
      status: "fail",
      leadId: job?.data?.leadId,
      message: `Job ${job?.id} failed: ${err.message}`,
    });
  });
  ```

  **File 3: src/acquisition/run.ts**

  CLI enqueue script — short-lived process. Reads status=New leads, bulk-enqueues, then exits. dotenv MUST be loaded first.

  ```typescript
  // MUST be the very first import — loads DATABASE_URL and REDIS_URL before singletons initialize
  import { config } from "dotenv";
  config({ path: ".env.local" });

  // All other imports AFTER dotenv
  import { db } from "@/db/index";
  import { leads } from "@/db/schema";
  import { acquisitionQueue } from "@/workers/queues";
  import { logger } from "@/lib/logger";
  import { eq } from "drizzle-orm";

  async function main(): Promise<void> {
    const newLeads = await db
      .select({ id: leads.id, url: leads.url })
      .from(leads)
      .where(eq(leads.status, "New"));

    if (newLeads.length === 0) {
      logger.info({ stage: "acquire", status: "ok", message: "No New leads to enqueue" });
      await acquisitionQueue.close();
      return;
    }

    // addBulk is atomic — all jobs added or none (prevents partial enqueue on Redis error)
    await acquisitionQueue.addBulk(
      newLeads.map((lead) => ({
        name: "acquire",
        data: { leadId: lead.id, url: lead.url },
      })),
    );

    logger.info({
      stage: "acquire",
      status: "ok",
      message: `Enqueued ${newLeads.length} acquisition jobs`,
    });

    // REQUIRED: close the Redis connection so the process exits cleanly
    await acquisitionQueue.close();
  }

  main().catch((err) => {
    logger.error({ stage: "acquire", status: "fail", message: `Fatal: ${String(err)}` });
    process.exit(1);
  });
  ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI process → Redis (BullMQ Queue) | Job payloads {leadId, url} written to Redis; Redis is local/trusted in dev |
| Redis (BullMQ) → Worker process | Job data read from Redis by worker; same trust as Queue write |
| Worker process → PostgreSQL | Lead status updates and manufacturer_pages inserts |
| Worker process → External manufacturer websites | Outbound Playwright browser sessions to untrusted websites |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-11 | Tampering | BullMQ job payload {leadId, url} could be crafted with arbitrary URL if Redis is accessible to external actor | accept | Redis is on localhost in dev; in production, Redis should be firewalled to allow only internal connections. url originates from leads.url which was written by the Chemexcil scraper from a public directory. No user-controlled input reaches the job payload in Phase 3. |
| T-03-12 | Denial of Service | npm run acquire enqueues thousands of jobs simultaneously, overwhelming the worker or Playwright | mitigate | concurrency: 3 limits simultaneous Playwright sessions. BullMQ naturally queues excess jobs. Each job has requestHandlerTimeoutSecs: 60 to prevent hangs. |
| T-03-13 | Repudiation | Worker process killed with SIGKILL (unhandled) leaves in-flight jobs in "active" state in BullMQ | accept | BullMQ marks jobs as stalled after a timeout and re-enqueues them. The stall interval is configurable; default is sufficient for dev use. In production, configure stalled job monitoring. |
| T-03-14 | Information Disclosure | Structured log lines in workers/index.ts and acquisition/run.ts may include lead IDs and URLs | accept | Logs are stdout on a developer machine; no PII in leadId or manufacturer URLs. Chemical company names from a public directory are not sensitive. |
</threat_model>
